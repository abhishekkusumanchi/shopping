-- User_Credentials TableCREATE TABLE User_Credentials (    user_name VARCHAR(255) PRIMARY KEY,    email VARCHAR(255) UNIQUE NOT NULL,    password_hash CHAR(64) NOT NULL);
-- Address_info TableCREATE TABLE Address_info (	address_id SERIAL PRIMARY KEY,    user_name VARCHAR(255) REFERENCES User_Credentials(user_name),    customer_name VARCHAR(255) NOT NULL,    mobile VARCHAR(15) NOT NULL,    email VARCHAR(255) REFERENCES User_Credentials(email),    pincode INT NOT NULL,    address VARCHAR(255));
-- ProductCategories TableCREATE TABLE ProductCategories (    prct_id SERIAL PRIMARY KEY,    prct_title VARCHAR(100));
-- HSNCodes TableCREATE TABLE HSNCodes (    hsnc_id SERIAL PRIMARY KEY,    hsnc_hsncode VARCHAR(20),    hsnc_gstc_percentage NUMERIC(20,2));
-- Products TableCREATE TABLE Products (    prod_id SERIAL PRIMARY KEY,    prod_title VARCHAR(100),    prod_prct_id INT REFERENCES ProductCategories(prct_id),    prod_hsnc_id INT REFERENCES HSNCodes(hsnc_id),    prod_brand VARCHAR(100),    prod_image VARCHAR(100),    prod_shipping_charge NUMERIC,    prod_base_price NUMERIC);
-- ProductStock TableCREATE TABLE ProductStock (    prod_id INT REFERENCES Products(prod_id),    prod_stock INT);
-- UnserviceableRegions TableCREATE TABLE UnserviceableRegions (    pincode INT,    prod_id INT,    PRIMARY KEY (pincode, prod_id));
-- ShippingDiscount TableCREATE TABLE ShippingDiscount (    total_price_from NUMERIC,    total_price_to NUMERIC,    discount INT,    CONSTRAINT pk_shipping_discount PRIMARY KEY (total_price_from, total_price_to));
-- DiscountCoupon TableCREATE TABLE DiscountCoupon (    dcpn_id SERIAL PRIMARY KEY,    dcpn_code VARCHAR(10) UNIQUE,    dcpn_value NUMERIC,    no_of_cpn INT,    dcpn_min_val NUMERIC,    expire_date DATE );
-- Orders TableCREATE TABLE Orders (    order_ID SERIAL PRIMARY KEY,    user_name VARCHAR(255) REFERENCES User_Credentials(user_name),    order_date DATE,    order_total DECIMAL,    dcpn_code VARCHAR(10) REFERENCES DiscountCoupon(dcpn_code),	address_id INT REFERENCES  Address_info(address_id));
-- Order_Product TableCREATE TABLE Order_Product (    order_ID INT REFERENCES Orders(order_ID),    product_id INT REFERENCES Products(prod_id),    quantity INT,    price DECIMAL,    total_gst DECIMAL,    CONSTRAINT fk_order_product_order_id FOREIGN KEY (order_ID) REFERENCES Orders(order_ID),    CONSTRAINT fk_order_product_product_id FOREIGN KEY (product_id) REFERENCES Products(prod_id));
